can_be_monastic_assistant = {
	basic_is_available_ai = yes
	NOT = { has_trait = incapable }
	is_capable_adult = yes
	is_imprisoned = no
	is_at_war_with_liege = no
}
faith_gender_check_trigger = {
	#scope:actor.faith = { has_allowed_gender_for_clergy = yes }
	trigger_if = { 
		limit = {
			scope:actor.faith = { has_doctrine = doctrine_gender_male_dominated }
			
		}
		is_female = no
	}
	trigger_if = { 
		limit = {
			scope:actor.faith = { has_doctrine = doctrine_gender_female_dominated }
			
		}
		is_male = no
	}
}
has_monastic_position_trigger = {
	OR = {
		any_relation = {
			type = monastery_prior
			this = scope:recipient
		}
		any_relation = {
			type = monastery_cantor
			this = scope:recipient
		}
		any_relation = {
			type = monastery_elder
			this = scope:recipient
		}
		any_relation = {
			type = monastery_cellarer
			this = scope:recipient
		}
		any_relation = {
			type = monastery_advocate
			this = scope:recipient
		}
	}
}
#used to prevent kicking in "kick_from_court_validity_trigger"
holds_monastic_position_trigger = { 
	OR = {
		has_character_modifier = mymo_monastic_prior
		has_character_modifier = mymo_monastic_cantor
		has_character_modifier = mymo_monastic_elder
		has_character_modifier = mymo_monastic_cellarer
		has_character_modifier = mymo_monastic_advocate
	}
}
