	trigger_if = { 
		limit = {
			scope:actor.faith = { has_doctrine = doctrine_gender_male_dominated }
			
		}
		is_female = no
	}
	trigger_if = { 
		limit = {
			scope:actor.faith = { has_doctrine = doctrine_gender_female_dominated }
			
		}
		is_male = no
	}
}
